---
title: Little Thing
date: 2022-12-06 19:45 -0800
tags:
- Peaceful
---
import { Body, L, Lg } from "@features/poem";
import { Q } from "@features/ui";

<Body>
<Lg>
<L>A small thing sat in the dirt</L>
<L>legs splayed, hands by its sides.</L>
<L><Q>I am tall</Q> it said</L>
<L>and stood to prove its point.</L>
</Lg>

<Lg>
<L>Nothing but the sky around</L>
<L>so vast the earth surround.</L>
<L>could hear and scarcely see</L>
<L>the little speck.</L>
</Lg>

<Lg>
<L>The sky knelt a little lower</L>
<L>to observe this would-be challenger.</L>
<L><Q>I am small but I am strong</Q></L>
<L>so the little thing would amend.</L>
</Lg>
</Body>
