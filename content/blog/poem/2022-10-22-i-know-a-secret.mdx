import { Caesura, L, Lg } from "@features/poem";

<Lg>
<L><q>I know a secret</q></L>
<L>the creature said</L>
<L>He sat there on the cobblestones</L>
<L>in the darkness, just out of sight</L>
<L>of the yellow sodium light</L>
</Lg>

<Lg>
<L>We sat there for a while</L>
<L>Me on the bench, him by the sea</L>
<L>The wind not much</L>
<L>by the waterfront</L>
</Lg>

<Lg>
<L>The creature spoke again</L>
</Lg>

> <Lg>
> <L>a few years back,</L>
> <L>I was sitting not far from here</L>
> <L>in the darkness, just out of sight</L>
> <L>a man raped a girl under cover of the night</L>
> </Lg>

> <Lg>
> <L>I know a secret,</L>
> <L>for I see very well,</L>
> <L>even under<Caesura />the night.</L>
> <L>These eyes catch many things</L>
> <L>I will never tell.</L>
> </Lg>

> <Lg>
> <L>The truth came out</L>
> <L>decades later at the trial</L>
> <L>Your courts<Caesura />are not like ours</L>
> <L>We have eyes<Caesura />which catch</L>
> <L>The beating of the heart</L>
> <L>Whispers<Caesura />inside the throat</L>
> </Lg>

> <Lg>
> <L>I saw an old man ill and dying</L>
> <L>too rich and sick for jail</L>
> <L>When he was coughing sputum up</L>
> <L>I was sitting there</L>
> <L>I know a secret</L>
> <L>and I will never tell</L>
> </Lg>
