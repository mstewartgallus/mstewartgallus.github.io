import { Body, Caesura, L, Lg } from "@features/poem";
import { Q } from "@features/ui";

<Body>
<Lg>
<L><Q>I know a secret</Q></L>
<L>the creature said</L>
<L>He sat there on the cobblestones</L>
<L>in the darkness, just out of sight</L>
<L>of the yellow sodium light</L>
</Lg>

<Lg>
<L>We sat there for a while</L>
<L>Me on the bench, him by the sea</L>
<L>The wind not much</L>
<L>by the waterfront</L>
</Lg>

<Lg>
<L>The creature spoke again</L>
<L><Q>a few years back,</Q></L>
<L><Q>I was sitting not far from here</Q></L>
<L><Q>in the darkness, just out of sight</Q></L>
<L><Q>a man raped a girl under cover of the night</Q></L>
</Lg>

<Lg>
<L><Q>I know a secret,</Q></L>
<L><Q>for I see very well,</Q></L>
<L><Q>even under<Caesura />the night.</Q></L>
<L><Q>These eyes catch many things</Q></L>
<L><Q>I will never tell.</Q></L>
</Lg>

<Lg>
<L><Q>The truth came out</Q></L>
<L><Q>decades later at the trial</Q></L>
<L><Q>Your courts<Caesura />are not like ours</Q></L>
<L><Q>We have eyes<Caesura />which catch</Q></L>
<L><Q>The beating of the heart</Q></L>
<L><Q>Whispers<Caesura />inside the throat</Q></L>
</Lg>

<Lg>
<L><Q>I saw an old man ill and dying</Q></L>
<L><Q>too rich and sick for jail</Q></L>
<L><Q>When he was coughing sputum up</Q></L>
<L><Q>I was sitting there</Q></L>
<L><Q>I know a secret</Q></L>
<L><Q>and I will never tell</Q></L>
</Lg>

</Body>
