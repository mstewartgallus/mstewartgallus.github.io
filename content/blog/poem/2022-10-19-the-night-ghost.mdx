import { Body, Caesura, L, Lg } from "@features/poem";

<Body>
    <Lg>
    <L>The bungler in the grip of the demon held</L>
    <L>To his crypt by strange means is compelled</L>
    <L>And sleeping as a corpse in darkness rotting</L>
    <L>He dreams forth the red eyed thing</L>
    <L>So rises from the sleeper's breath</L>
    <L>Comes the Night Ghost<Caesura />who brings death!</L>
    </Lg>

    <Lg>
    <L>The Night Ghost climbs, as a shadow leaping, overtop the manor wall</L>
    <L>And through an open window goes, as a chilling fog</L>
    <L>To the room of the sleeper he creeps</L>
    <L>And as a swarm of bugs, overtop her stomach crawls</L>
    <L>And on her chest, as a choking darkness sits</L>
    </Lg>

    <Lg>
    <L>'Til she wakes with a gasp</L>
    <L>The Night Ghost; by the window watching, giant as a shadow standing</L>
    <L>She claps &lt;clap sound effect&gt; turning on the light</L>
    <L>And The Night Ghost, as the darkness in the day, vanishes from all sight.</L>
    </Lg>
</Body>
